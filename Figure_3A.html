library(tidyverse)
library(patchwork)

filepath <- here::here("extdata/Figure_3A.csv")

enzymes <- read.csv(filepath) %>% dplyr::filter(date != "5_31_17") %>%
  format_AvoidData(day.correct = TRUE) %>%
  mutate(strain = factor(strain, levels = c("OP50", "JUb39")),
         genotype = factor(genotype, levels = c("N2", "cat-2", "tph-1", "tdc-1", "tbh-1")))
         #plate = factor(seq(1:nrow(.))))

p1 <- enzymes %>% 
  plot_plasticityIndex(xvar = strain) +
    scale_color_plot('grey-blue', drop = TRUE) +
  guides(color = FALSE) + labs(title = "1-oct") +
  facet_grid(.~genotype) +
  theme_my_ppt

p2 = enzymes %>% 
  plot_ChemoIndex(xvar = "strain", plot.pos = FALSE) +
  facet_grid(.~genotype, drop = TRUE) +
  scale_color_plot('grey-blue', drop = FALSE) +
  guides(fill = FALSE) + labs(title = "1-oct") +
  coord_cartesian(ylim = c(-1.1, 0.5)) +
  theme_my_ppt

p1 + p2

sciplot::bargraph.CI(data = enzymes,
                     x.factor = genotype,
                     response = rel.Logit,
                     group = strain,
                     #ylim = c(-.5,3),
                     ylim = c(-.5,3),
                     col=c("#827E7E","#484CC7"),
                     err.width = 0.05
                     )
